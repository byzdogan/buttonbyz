## Install

```bash
npm install --save buttonbyz
```

## Usage

```jsx
import React, { Component } from 'react'

import { Button } from 'buttonbyz'
import 'buttonbyz/dist/index.css'

class Example extends Component {
  render() {
    return <Button>Click</Button>
  }
}
```
